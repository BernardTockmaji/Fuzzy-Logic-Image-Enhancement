5- Calculate fuzzy membership values muC1 and muC2 using equations 2 and 4 & Calculate the enhanced intensity values using equations 3 and 5

In [ ]:
def enhance(X):
    if X in C1:
        # Equation 2
        muC1 = (1 - (M - X)) / M
        # Equation 3
        X = X + (muC1 * K)
        if X < 0:
            X = 0
    elif X in C2:
        # Equation 4
        muC2 = (E - X) / (E - M)
        # Equation 5
        X = (X * muC2) + (E - (muC2 * K)) - K
        if X > 255:
            X = 255
    return X
